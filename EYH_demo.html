Simple Plot

library(ggplot2)
## Warning: package 'ggplot2' was built under R version 3.3.2
recent_data <- gapminder[gapminder$year == 2007,]

ggplot(recent_data, 
       aes(x = gdpPercap, y = lifeExp)) + 
    geom_point() + 
    theme_bw()

ggplot(recent_data, 
       aes(x = continent, y = lifeExp)) + 
    geom_bar(stat = "summary", fun.y = "mean") + 
    theme_bw()

ggplot(recent_data, 
       aes(x = continent, y = lifeExp)) + 
    geom_boxplot() + 
    theme_bw()
